#include <bits/stdc++.h>
using namespace std;

int main() {
	// your code goes here
	int n,x;cin>>n>>x;
	int modd=1e9+7;
	vector<int>coins(n);
	for(int i=0;i<n;i++){
		cin>>coins[i];
	}
	vector<int>dp(x+1);
	dp[0]=1;
	for(auto coin:coins){
		for(int i=coin;i<=x;i++){
			dp[i]=(dp[i]+dp[i-coin])%modd;
		}
	}
	cout<<dp[x];
	return 0;
}
